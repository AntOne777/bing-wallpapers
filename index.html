fetch('data.json?t=' + Date.now()) .then(r => r.json()) .then(data => { Object.values(data).forEach(item => { const card = document.createElement('div'); card.className = 'card'; // Защита от undefined: если данных нет, ставим значения по умолчанию const title = item.title || "Без названия"; const date = item.date || ""; const markets = (item.markets || []).join(' • '); card.innerHTML = `
${title}
${date} ${markets}
`; const img = card.querySelector('img'); img.onload = () => img.classList.add('loaded'); if(img.complete) img.classList.add('loaded'); card.querySelector('.card-overlay').onclick = (e) => { e.preventDefault(); lbImg.src = item.url; // ИМЯ ФАЙЛА БЕЗ ДАТЫ lbl.textContent = `${item.img_id}.jpg`; lb.classList.add('active'); dlBtn.onclick = async () => { dlBtn.textContent = "Загрузка..."; const r = await fetch(item.url); const b = await r.blob(); if(currentBlobUrl) URL.revokeObjectURL(currentBlobUrl); currentBlobUrl = URL.createObjectURL(b); const a = document.createElement('a'); a.href = currentBlobUrl; a.download = lbl.textContent; // Здесь будет только ID.jpg a.click(); dlBtn.textContent = "Скачать UHD"; }; }; document.getElementById('gallery').append(card); }); });
